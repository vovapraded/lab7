Формат команд
<команда> <аргумент>


"info" - "Выводит информацию о коллекции"
"show" - "Выводит коллекцию"
"insert" - "Добавить элемент"
"update" - "Обновить элемент"
"remove_key" - "Удалить элемент"
"clear" - "Очистить коллекцию"
"save" - "Сохранить коллекцию"
"execute_script" - "Выполнить скрипт из файла"
"exit" - "Завершение работы"
"remove_greater" - "Удалить элементы с ценой больше указанной"
"replace_if_greater" - "Заменить элемент, если цена введённого выше"
"remove_greater_key" - "Удалить элементы с id больше заданного числа"
"average_of_price" - "Вывести среднюю цену"
"filter_less_than_venue" - "Вывести элементы у которых venue.capacity меньше данного числа"
"print_descending" - "Вывести коллекцию в обратном порядке"
